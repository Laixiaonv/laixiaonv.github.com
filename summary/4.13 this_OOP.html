关于this的指向问题：this是在函数真正被调用执行的时候确定的

  • 构造函数 
    						function Foo(){
							this.name = "xiaohei";
							this.year = 1988;
							console.log(this);     //此时this指向的是new出来的对象
						}						//注意：仅限new的情况
						var f1 = new Foo();
						alert(f1.name);			//xiaohei
						alert(f1.year);			//1988
					若不new一个对象，直接调用Foo(),此时指向的是window
  • 函数作为对象的一个属性
    若函数作为对象的一个属性时，并且作为对象的一个属性被调用时，函数中的this指向该对象
    						var obj ={
							x:10,
							fn:function(){
								console.log(this);     //返回obj
								console.log(this.x);   //10
							}
						};
						obj.fn();    
						简单来说就是“ . ”的前面是什么，指向的就是什么
						
						下面是不作为属性被调用的情况
						即 var fn1 = obj.fn;
								fn1();
							console.log(this);			//返回window
							console.log(this.x);		//undefined
						fn被赋值给fn1，没有作为obj的一个属性被调用，此时this就是window
  • 全局 & 普通调用
    全局环境下，this永远指向window
    普通函数在调用时，this也指向window
    					var obj = {
						x:10,
						fn:function(){
							function f(){
								console.log(this);		//window
								console.log(this.x);		//undefined
							}						//在该情况下，函数虽然是在obj.fn的内部定义，但仍然是一个普通的函数
							f();
						} 					
					}
					obj.fn();
  • 函数用call或者apply调用，this的值就取传入的对象的值 
    					var obj ={
						x:10
					}
					var fn = function(){
						console.log(this);    //obj
						console.log(this.x);	//10
					}
					fn.apply(obj);
					若fn(obj);则返回的是window和undefined

    • 这是一个终极而且简直了的例子

    					function Foo(){
						getName = function(){     //这是一个全局的变量
					  		console.log(1);	
					 	}
					  	return this;
					}
					Foo.getName = function(){
					  	console.log(2);
					}
					Foo.prototype.getName = function(){
					 	console.log(3);
					}
					var getName = function(){
					  	console.log(4);
					}
					function getName(){
					  	console.log(5);
					} 
					Foo.getName();  	 		//2
					getName();       			//4
					Foo().getName(); 			//1 
					getName();				//1  
					new Foo.getName();		//2 
					new Foo().getName();		//3   
					这里new先找到了Foo(),因为这玩意带括号，所以会先执行new Foo(),但是仅限这种带括号的情况
					new new Foo().getName();	//3
