					function dateBook(year,month){
						var days = new Date(year,month,0).getDate();		//获取这个月的天数
						var firstDay = new Date(year + "-" + month + "-1").getDay();		//获取这个月第一天是星期几
						var col;		//var 一个变量，存放日历显示的行数
						if(firstDay == "0"){		//因为在国外星期天默认为“0”而不是“7”，所以我们要对第一天是星期天的情况做一个处理
							firstDay = "7";
							col = Math.ceil((days+firstDay)/7);		//因为我们在计算一共有几行的时候要算上前面空出的那几天，也就是第一天是星期几，
																	前面就空出相应的天数
						}else{
							col =  Math.ceil((days+firstDay) / 7 ); 
						}
						for(var i=0 ; i< col ; i ++){			//根据天数创建行
							var tr = document.createElement("tr");
								table.appendChild(tr);
							for(var j = 1 ; j<=7 ; j++){		//一个创建col*7个单元格
								var td = document.createElement("td");
									//判断,当数值小于1或者大于一个月的天数时,那几个单元格都清空
									if(i*7+j-firstDay + 1<=0 || i*7+j-firstDay + 1>days){
										td.innerText = "";
									}else{
										td.innerText = i*7 + j -firstDay + 1;	
									}
								tr.appendChild(td);	
							} 
						}
						//周六、日的天数显示样式为红色
						for(var i = 0;i < col;i++){
							table.rows[i].cells[5].className = "color"; 
							table.rows[i].cells[6].className = "color"; 
						}
					}
